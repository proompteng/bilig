Why it exists

Formulas are product logic. Treat them like code.

Use bilig when a spreadsheet model should live inside a service, queue worker, tool call, or serverless route instead of a user-facing spreadsheet tab.

01 / create

Build sheets from arrays, JSON records, or service data.

Examples cover invoice totals, budget variance, fulfillment capacity, quote approval, and subscription MRR.

02 / edit

Write inputs and structural changes through a workbook API.

No screen scraping. No hidden UI state. The changed workbook is the object your code already owns.

03 / read

Recalculate and read the value that was actually produced.

Agent tools can return evidence after each write instead of guessing from a screenshot or stale cache.

04 / persist

Serialize the WorkPaper document and restore it later.

The persistence guides show JSON state behind SQLite, Postgres, and object storage adapters.

The public examples are TypeScript, including examples/headless-workpaper and examples/serverless-workpaper-api. 

import {
  WorkPaper,
  createWorkPaperFromDocument,
  exportWorkPaperDocument,
  parseWorkPaperDocument,
  serializeWorkPaperDocument,
} from "@bilig/headless";

type NumericCell = { value: number };

const workbook = WorkPaper.buildFromSheets({
  Revenue: [
    ["Region", "Customers", "ARPA", "Revenue"],
    ["West", 20, 1200, "=B2*C2"],
    ["East", 30, 250, "=B3*C3"],
    ["Central", 18, 300, "=B4*C4"],
  ],
  Summary: [
    ["Metric", "Value"],
    ["Total revenue", "=SUM(Revenue!D2:D4)"],
  ],
});

function numberValue(cell: unknown): number {
  if (typeof cell === "object" && cell !== null && typeof (cell as NumericCell).value === "number") {
    return (cell as NumericCell).value;
  }
  throw new Error(`Expected numeric cell value, got ${JSON.stringify(cell)}`);
}

const summary = workbook.getSheetId("Summary");
const revenue = workbook.getSheetId("Revenue");
if (summary === undefined || revenue === undefined) {
  throw new Error("Missing sheet");
}

const before = numberValue(workbook.getCellValue({ sheet: summary, row: 1, col: 1 }));
workbook.setCellContents({ sheet: revenue, row: 1, col: 1 }, 32);

const saved = serializeWorkPaperDocument(exportWorkPaperDocument(workbook, { includeConfig: true }));
const restored = createWorkPaperFromDocument(parseWorkPaperDocument(saved));
const restoredSummary = restored.getSheetId("Summary");
if (restoredSummary === undefined) {
  throw new Error("Missing restored Summary sheet");
}

const after = numberValue(restored.getCellValue({ sheet: restoredSummary, row: 1, col: 1 }));
const verified = before === 36900 && after === 51300 && saved.length > 0;
console.log({ before, after, bytes: saved.length, verified });
